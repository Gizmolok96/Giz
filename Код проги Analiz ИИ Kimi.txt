# Updated Code for Automatic Match Results Fetching and ML Model Training

## Features:
- Automatic fetching of complete match results.
- Machine Learning model training capabilities based on completed matches.

## Code:

import requests
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
import pandas as pd

class MatchAnalyzer:
    def __init__(self):
        self.matches = []
        self.model = None

    def fetch_matches(self):
        response = requests.get('https://api.example.com/matches')  # URL for fetching match data
        if response.status_code == 200:
            self.matches = response.json()  # Assuming the API returns JSON data
        else:
            print("Failed to fetch matches")

    def prepare_data(self):
        data = []
        for match in self.matches:
            # Extract relevant features from match data
            data.append({
                'feature1': match['feature1'],
                'feature2': match['feature2'],  # Example features
                'result': match['result']
            })
        return pd.DataFrame(data)

    def train_model(self):
        df = self.prepare_data()
        X = df[['feature1', 'feature2']]  # Features for the model
        y = df['result']  # Target variable
        X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
        self.model = RandomForestClassifier()
        self.model.fit(X_train, y_train)
        print("Model trained successfully")

    def predict(self, new_data):
        if self.model:
            return self.model.predict([new_data])
        else:
            print("Model not trained yet")
