def cache_api_request(ttl_seconds=1800):
    """Декоратор для кэширования API запросов"""
    cache = {}

    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            key = hashlib.md5(
                f"{func.__name__}:{str(args)}:{str(kwargs)}".encode()
            ).hexdigest()

            now = time.time()

            if key in cache:
                result, timestamp = cache[key]
                if now - timestamp < ttl_seconds:
                    return result

            result = func(*args, **kwargs)
            cache[key] = (result, now)
            return result

        wrapper.clear_cache = lambda: cache.clear()
        return wrapper

    return decorator


@dataclass
class TeamFeatures:
    """Структура для хранения признаков команды"""
    # Базовая статистика
    points_per_game: float = 1.0
    win_rate: float = 0.33
    draw_rate: float = 0.33
    loss_rate: float = 0.34

    # Атака
    avg_goals_for: float = 1.3
    avg_shots: float = 12.0
    avg_shots_on_target: float = 4.0
    finishing_efficiency: float = 1.0
    xg_for_avg: float = 1.3

    # Оборона
    avg_goals_against: float = 1.3
    xg_against_avg: float = 1.3
    defensive_efficiency: float = 1.0

    # Владение и контроль
    avg_possession: float = 50.0
    avg_corners: float = 5.0

    # Форма (последние 5 матчей)

@dataclass
class MatchFeatures:
    """Полный набор признаков для матча"""
    home: TeamFeatures
    away: TeamFeatures

    # H2H статистика
    h2h_home_wins: int = 0
    h2h_draws: int = 0
    h2h_away_wins: int = 0
    h2h_avg_goals: float = 2.5

    # Контекст лиги
    league_avg_goals: float = 2.5
    league_home_advantage: float = 1.4

    def to_list(self) -> List[float]:
        """Общий вектор признаков для ML модели"""
        features = []
        features.extend(self.home.to_list())
        features.extend(self.away.to_list())
        features.extend([
            self.h2h_home_wins / max(self.total_h2h, 1),
            self.h2h_draws / max(self.total_h2h, 1),
            self.h2h_away_wins / max(self.total_h2h, 1),
            self.h2h_avg_goals,
            self.league_avg_goals,
            self.league_home_advantage,
        ])
        return features

    @property
    def total_h2h(self) -> int:
        return self.h2h_home_wins + self.h2h_draws + self.h2h_away_wins

    def to_dict(self) -> Dict:
        """Сохранение в БД"""
        return {
            'home_features': asdict(self.home),
            'away_features': asdict(self.away),
            'h2h_home_wins': self.h2h_home_wins,
            'h2h_draws': self.h2h_draws,
            'h2h_away_wins': self.h2h_away_wins,
            'h2h_avg_goals': self.h2h_avg_goals,
            'league_avg_goals': self.league_avg_goals,
            'league_home_advantage': self.league_home_advantage,
        }

    @classmethod
    def from_dict(cls, data: Dict) -> 'MatchFeatures':
        """Загрузка из БД"""
        home = TeamFeatures(**data['home_features'])
        away = TeamFeatures(**data['away_features'])

        return cls(
            home=home,
            away=away,
            h2h_home_wins=data.get('h2h_home_wins', 0),
            h2h_draws=data.get('h2h_draws', 0),
            h2h_away_wins=data.get('h2h_away_wins', 0),
            h2h_avg_goals=data.get('h2h_avg_goals', 2.5),
            league_avg_goals=data.get('league_avg_goals', 2.5),
            league_home_advantage=data.get('league_home_advantage', 1.4),
        )

        conn = sqlite3.connect(str(self.db_path))
        cursor = conn.cursor()

        # Матчи
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS matches (
                id INTEGER PRIMARY KEY,

                date TEXT,
                home_team TEXT,

                home_team_id INTEGER,

                away_xg REAL,

            )
        """)

        # История прогнозов с полными признаками
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS prediction_history (
                id INTEGER PRIMARY KEY,
                match_id INTEGER,
                home_team TEXT,
                away_team TEXT,
                match_date TEXT,

                -- Сохраняем полные признаки для ML (JSON)
                features_json TEXT,

                -- Прогнозы
                predicted_home_win REAL,
                predicted_draw REAL,
                predicted_away_win REAL,
                predicted_score TEXT,

                model_version TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                resolved_at TIMESTAMP
            )
        """)

        # Статистика точности модели
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS model_performance (
                id INTEGER PRIMARY KEY,
                model_version TEXT,
                training_date TIMESTAMP,
                total_matches INTEGER,
                accuracy REAL,
                precision_home REAL,
                precision_draw REAL,
                precision_away REAL,
                features_used TEXT
            )
        """)

        # Настройки
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS settings (
                key TEXT PRIMARY KEY,
                value TEXT
            )
        """)

        conn.commit()
        conn.close()

    def get_training_data(self, min_confidence: float = 0.0) -> List[Dict]:
        """Получить данные для обучения ML (только с известными результатами)"""
        conn = sqlite3.connect(str(self.db_path))
        conn.row_factory = sqlite3.Row
        cursor = conn.cursor()

        cursor.execute("""
            SELECT * FROM prediction_history 
            WHERE actual_result IS NOT NULL
            AND features_json IS NOT NULL
            AND (ml_confidence IS NULL OR ml_confidence >= ?)
            ORDER BY created_at DESC
        """, (min_confidence,))

        rows = cursor.fetchall()
        conn.close()

        result = []
        for row in rows:
            data = dict(row)
            try:
                data['features'] = MatchFeatures.from_dict(json.loads(data['features_json']))
                result.append(data)
            except:
                continue

        return result

    def update_match_result(self, match_id: int, home_goals: int, away_goals: int):
        """Обновить результат матча и связанный прогноз"""
        conn = sqlite3.connect(str(self.db_path))
        cursor = conn.cursor()

        if home_goals > away_goals:
            actual_result = 'home'
        elif home_goals < away_goals:
            actual_result = 'away'
        else:
            actual_result = 'draw'

        # Получаем предсказание
        cursor.execute("""
            SELECT predicted_home_win, predicted_draw, predicted_away_win, ml_predicted_class
            FROM prediction_history WHERE match_id = ?
        """, (match_id,))

        row = cursor.fetchone()
        if row:
            pred_home, pred_draw, pred_away, ml_class = row

            # Определяем предсказанный класс
            max_prob = max(pred_home, pred_draw, pred_away)
            if max_prob == pred_home:
                predicted_class = 1
            elif max_prob == pred_draw:
                predicted_class = 0
            else:
                predicted_class = 2

            # Кодируем фактический результат
            actual_class = {'draw': 0, 'home': 1, 'away': 2}[actual_result]

            # Проверяем точность
            correct = (predicted_class == actual_class)

            cursor.execute("""
                UPDATE prediction_history 
                SET actual_home_goals = ?, actual_away_goals = ?, 
                    actual_result = ?, prediction_correct = ?, resolved_at = CURRENT_TIMESTAMP
                WHERE match_id = ?
            """, (home_goals, away_goals, actual_result, correct, match_id))

            conn.commit()

        conn.close()

    def get_model_stats(self) -> Dict:
        """Получить статистику модели"""
        conn = sqlite3.connect(str(self.db_path))
        conn.row_factory = sqlite3.Row
        cursor = conn.cursor()

        # Общая точность
        cursor.execute("""
            SELECT 
                COUNT(*) as total,
                SUM(CASE WHEN prediction_correct = 1 THEN 1 ELSE 0 END) as correct,
                AVG(ml_confidence) as avg_confidence
            FROM prediction_history
            WHERE actual_result IS NOT NULL
        """)

        row = cursor.fetchone()
        stats = {
            'total_predicted': row['total'] or 0,
            'total_correct': row['correct'] or 0,
            'accuracy': (row['correct'] / row['total'] * 100) if row['total'] else 0,
            'avg_ml_confidence': row['avg_confidence'] or 0,
        }

        # По исходам
        cursor.execute("""
            SELECT actual_result, COUNT(*) as count,
                   SUM(CASE WHEN prediction_correct = 1 THEN 1 ELSE 0 END) as correct
            FROM prediction_history
            WHERE actual_result IS NOT NULL
            GROUP BY actual_result
        """)

        stats['by_result'] = {row['actual_result']: {
            'total': row['count'],
            'correct': row['correct'],
            'accuracy': row['correct'] / row['count'] * 100 if row['count'] else 0
        } for row in cursor.fetchall()}

        conn.close()
        return stats

    def get_prediction_history(self, limit: int = 50) -> List[Dict]:
        """История прогнозов (для окна истории)"""
        conn = sqlite3.connect(str(self.db_path))
        conn.row_factory = sqlite3.Row
        cursor = conn.cursor()

        cursor.execute("""
            SELECT *,
                   CASE 
                       WHEN actual_home_goals IS NOT NULL AND actual_away_goals IS NOT NULL
                       THEN actual_home_goals || ':' || actual_away_goals
                       ELSE NULL
                   END AS actual_score
            FROM prediction_history
            ORDER BY created_at DESC
            LIMIT ?
        """, (limit,))

        rows = cursor.fetchall()
        conn.close()
        return [dict(r) for r in rows]

    def auto_update_finished_matches(self, matches: List[Dict]):
        """
        Автоматически обновить счета завершенных матчей по данным API.
        Для каждого матча:
        - Если статус окончен (finished/ended) и есть счет, обновляем таблицу matches
        - Обновляем prediction_history через update_match_result
        """
        conn = sqlite3.connect(str(self.db_path))
        cursor = conn.cursor()

        updated = 0

        for m in matches:
            try:
                match_id = m.get('id')
                status = str(m.get('status', '')).lower()
                home_goals = m.get('homeResult')
                away_goals = m.get('awayResult')

                if match_id is None:
                    continue

                # считаем матч завершенным по статусу и наличию счета
                if status in ('finished', 'ended', 'ft', 'after_penalties') or (
                        home_goals is not None and away_goals is not None
                ):
                    # обновляем/вставляем в таблицу matches
                    cursor.execute("""
                        INSERT INTO matches (sstats_id, date, home_team, away_team,
                                             home_team_id, away_team_id, league, league_id,
                                             status, home_goals, away_goals, cached_at)
                        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP)
                        ON CONFLICT(sstats_id) DO UPDATE SET
                            status = excluded.status,
                            home_goals = excluded.home_goals,
                            away_goals = excluded.away_goals,
                            cached_at = CURRENT_TIMESTAMP
                    """, (
                        match_id,
                        m.get('date'),
                        (m.get('homeTeam') or {}).get('name', ''),
                        (m.get('awayTeam') or {}).get('name', ''),
                        (m.get('homeTeam') or {}).get('id'),
                        (m.get('awayTeam') or {}).get('id'),
                        (m.get('league') or {}).get('name', ''),
                        (m.get('league') or {}).get('id'),
                        status,
                        home_goals,
                        away_goals
                    ))

                    conn.commit()

                    if home_goals is not None and away_goals is not None:
                        # Обновляем связанный прогноз
                        self.update_match_result(match_id, int(home_goals), int(away_goals))
                        updated += 1
            except Exception as e:
                print(f"[DB] error in auto_update_finished_matches: {e}")
                continue

        conn.close()
        print(f"[DB] auto_update_finished_matches: updated {updated} matches")


class FootballMLModel:
    """Продвинутая ML модель для прогнозирования футбольных матчей"""

    def __init__(self):
        self.model = None
        self.scaler = StandardScaler() if SKLEARN_AVAILABLE else None
        self.model_dir = Path.home() / ".sstats_desktop" / "models"
        self.model_dir.mkdir(exist_ok=True)
        self.model_path = self.model_dir  # noqa: E305
        self.model_path = self.model_dir / "football_rf_v1.pkl"
        self.scaler_path = self.model_dir / "scaler_v1.pkl"
        self.metrics_path = self.model_dir / "metrics_v1.json"
        self.is_trained = False
        self.training_date = None
        self.accuracy = 0.0
        self.load_model()

    def load_model(self):
        """Загрузить сохраненную модель"""
        if not SKLEARN_AVAILABLE:
            return

        try:
            if self.model_path.exists():
                with open(self.model_path, 'rb') as f:
                    self.model = pickle.load(f)
                with open(self.scaler_path, 'rb') as f:
                    self.scaler = pickle.load(f)
                if self.metrics_path.exists():
                    with open(self.metrics_path, 'r') as f:
                        metrics = json.load(f)
                        self.accuracy = metrics.get('accuracy', 0)
                        self.training_date = metrics.get('date')
                self.is_trained = True
                print(f"ML Model loaded. Accuracy: {self.accuracy:.2f}%")
        except Exception as e:
            print(f"Error loading model: {e}")
            self.model = None
            self.is_trained = False

    def save_model(self, accuracy: float, report: Dict):
        """Сохранить модель и метрики"""
        if not SKLEARN_AVAILABLE or not self.model:
            return

        try:
            with open(self.model_path, 'wb') as f:
                pickle.dump(self.model, f)
            with open(self.scaler_path, 'wb') as f:
                pickle.dump(self.scaler, f)

            metrics = {
                'date': datetime.now().isoformat(),
                'accuracy': accuracy,
                'report': report,
                'feature_count': len(self.get_feature_names())
            }
            with open(self.metrics_path, 'w') as f:
                json.dump(metrics, f)

            self.training_date = metrics['date']
            self.accuracy = accuracy

        except Exception as e:
            print(f"Error saving model: {e}")

    def get_feature_names(self) -> List[str]:
        """Названия признаков для интерпретации"""
        home_features = [
            'home_ppg', 'home_win_rate', 'home_draw_rate', 'home_loss_rate',
            'home_avg_goals', 'home_shots', 'home_sot', 'home_finishing',
            'home_xg', 'home_goals_against', 'home_xg_against', 'home_def_eff',
            'home_possession', 'home_corners', 'home_recent_points',
            'home_recent_gf', 'home_recent_ga', 'home_trend', 'home_advantage', 'home_dummy'
        ]
        away_features = [f.replace('home_', 'away_').replace('advantage', 'disadvantage')
                         for f in home_features]
        h2h_features = ['h2h_home_ratio', 'h2h_draw_ratio', 'h2h_away_ratio',
                        'h2h_avg_goals', 'league_avg_goals', 'league_home_adv']

        return home_features + away_features + h2h_features

    def prepare_training_data(self, historical_data: List[Dict]) -> Tuple[np.ndarray, np.ndarray]:
        """Подготовить данные для обучения"""
        X = []
        y = []

        for item in historical_data:
            try:
                features = item['features']  # MatchFeatures object
                actual_result = item['actual_result']

                # Кодируем результат: 0=ничья, 1=хозяева, 2=гости
                result_map = {'draw': 0, 'home': 1, 'away': 2}
                y.append(result_map[actual_result])

                # Вектор признаков
                X.append(features.to_list())

            except Exception as e:
                print(f"Error processing training item: {e}")
                continue

        return np.array(X), np.array(y)

    def train(self, historical_data: List[Dict], test_size: float = 0.2) -> Dict:
        """Обучить модель на исторических данных с валидацией"""
        if not SKLEARN_AVAILABLE:
            return {'success': False, 'error': 'sklearn not available'}

        if len(historical_data) < 30:
            return {'success': False, 'error': f'Need 30+ matches, got {len(historical_data)}'}

        try:
            # Подготовка данных
            X, y = self.prepare_training_data(historical_data)

            if len(X) < 30:
                return {'success': False, 'error': 'Insufficient valid data after processing'}

            # Разделение на train/test
            X_train, X_test, y_train, y_test = train_test_split(
                X, y, test_size=test_size, random_state=42, stratify=y
            )

            # Масштабирование
            X_train_scaled = self.scaler.fit_transform(X_train)
            X_test_scaled = self.scaler.transform(X_test)

            # Обучение Random Forest
            self.model = RandomForestClassifier(
                n_estimators=200,
                max_depth=15,
                min_samples_split=5,
                min_samples_leaf=2,
                max_features='sqrt',
                class_weight='balanced',
                random_state=42,
                n_jobs=-1
            )

            self.model.fit(X_train_scaled, y_train)

            # Оценка
            y_pred = self.model.predict(X_test_scaled)
            accuracy = accuracy_score(y_test, y_pred) * 100

            # Кросс-валидация
            cv_scores = cross_val_score(self.model, X_train_scaled, y_train, cv=5)

            # Детальный отчет
            report = classification_report(y_test, y_pred,
                                           target_names=['Draw', 'Home', 'Away'],
                                           output_dict=True)

            # Важность признаков
            feature_importance = dict(zip(
                self.get_feature_names(),
                self.model.feature_importances_.tolist()
            ))
            top_features = sorted(feature_importance.items(),
                                  key=lambda x: x[1], reverse=True)[:10]

            # Сохранение
            self.save_model(accuracy, report)
            self.is_trained = True

            # Сохранение в БД
            db.save_model_performance('v1.0', accuracy, len(X), report)

            return {
                'success': True,
                'accuracy': accuracy,
                'cv_mean': cv_scores.mean() * 100,
                'cv_std': cv_scores.std() * 100,
                'train_size': len(X_train),
                'test_size': len(X_test),
                'report': report,
                'top_features': top_features
            }

        except Exception as e:
            return {'success': False, 'error': str(e)}

    def predict(self, features: MatchFeatures) -> Optional[Dict]:
        """Сделать предсказание для матча"""
        if not SKLEARN_AVAILABLE or not self.is_trained or not self.model:
            return None

        try:
            X = np.array([features.to_list()])
            X_scaled = self.scaler.transform(X)

            # Вероятности для каждого класса
            proba = self.model.predict_proba(X_scaled)[0]

            # Уверенность = максимальная вероятность
            confidence = max(proba) * 100

            # Если уверенность низкая, можно не доверять
            if confidence < 40:
                reliability = 'low'
            elif confidence < 60:
                reliability = 'medium'
            else:
                reliability = 'high'

            return {
                'draw': round(proba[0] * 100, 1),
                'home_win': round(proba[1] * 100, 1),
                'away_win': round(proba[2] * 100, 1),
                'predicted_class': int(np.argmax(proba)),
                'confidence': round(confidence, 1),
                'reliability': reliability,
                'model_accuracy': self.accuracy,
                'model_date': self.training_date
            }

        except Exception as e:
            print(f"Prediction error: {e}")
            return None

    def get_feature_importance(self) -> List[Tuple[str, float]]:
        """Получить важность признаков"""
        if not self.is_trained or not self.model:
            return []

        importance = list(zip(
            self.get_feature_names(),
            self.model.feature_importances_
        ))
        return sorted(importance, key=lambda x: x[1], reverse=True)


class AIAnalyzer:
    """AI анализатор с правильной ML интеграцией"""

    def __init__(self):
        self.league_context = self._init_league_context()
        self.ml_model = FootballMLModel()

    def _init_league_context(self) -> Dict:
        return {
            'premier_league': {'avg_goals': 2.7, 'home_advantage': 1.4, 'style': 'physical'},
            'la_liga': {'avg_goals': 2.5, 'home_advantage': 1.35, 'style': 'technical'},
            'serie_a': {'avg_goals': 2.4, 'home_advantage': 1.45, 'style': 'tactical'},
            'bundesliga': {'avg_goals': 3.0, 'home_advantage': 1.3, 'style': 'attacking'},
            'ligue_1': {'avg_goals': 2.3, 'home_advantage': 1.4, 'style': 'balanced'},
            'rpl': {'avg_goals': 2.2, 'home_advantage': 1.5, 'style': 'defensive'}
        }

    def extract_team_features(self, stats: Dict, is_home: bool,
                              league_context: Dict) -> TeamFeatures:
        """Извлечь структурированные признаки команды"""
        if not stats or 'matches' not in stats or not stats['matches']:
            # Дефолтные значения с небольшими корректировками
            defaults = TeamFeatures()
            if is_home:
                defaults.home_advantage = league_context['home_advantage']
                defaults.points_per_game *= league_context['home_advantage']
                defaults.avg_goals_for *= league_context['home_advantage']
            else:
                defaults.away_disadvantage = 0.85
                defaults.points_per_game *= 0.85
            defaults.is_home = is_home
            return defaults

        matches = stats['matches']
        total = len(matches)

        # Базовая статистика
        wins = sum(1 for m in matches if m.get('result') == 'W')
        draws = sum(1 for m in matches if m.get('result') == 'D')
        losses = total - wins - draws

        goals_for = sum(m.get('goals_for', 0) for m in matches)
        goals_against = sum(m.get('goals_against', 0) for m in matches)

        # xG
        xg_for_list = [m.get('xg_for') for m in matches if m.get('xg_for') is not None]
        xg_against_list = [m.get('xg_against') for m in matches if m.get('xg_against') is not None]

        xg_for_avg = sum(xg_for_list) / len(xg_for_list) if xg_for_list else goals_for / total
        xg_against_avg = sum(xg_against_list) / len(xg_against_list) if xg_against_list else goals_against / total

        # Эффективность
        finishing_eff = (goals_for / total) / xg_for_avg if xg_for_avg > 0 else 1.0
        defensive_eff = (goals_against / total) / xg_against_avg if xg_against_avg > 0 else 1.0

        # Средние показатели
        avg_possession = sum(m.get('possession', 50) for m in matches) / total
        avg_shots = sum(m.get('shots', 0) for m in matches) / total
        avg_sot = sum(m.get('shots_on_target', 0) for m in matches) / total
        avg_corners = sum(m.get('corners', 0) for m in matches) / total

        # Форма (последние 5)
        recent = matches[:5]
        recent_points = sum({'W': 3, 'D': 1, 'L': 0}.get(m.get('result'), 0) for m in recent)
        recent_gf = sum(m.get('goals_for', 0) for m in recent)
        recent_ga = sum(m.get('goals_against', 0) for m in recent)

        # Тренд
        trend = self._calculate_trend_value(recent)

        # Домашнее/выездное преимущество
        home_advantage = 1.0
        away_disadvantage = 1.0

        if is_home:
            home_matches = [m for m in matches if m.get('is_home')]
            if home_matches:
                home_ppg = sum({'W': 3, 'D': 1, 'L': 0}.get(m.get('result'), 0)
                               for m in home_matches) / len(home_matches)
                away_matches = [m for m in matches if not m.get('is_home')]
                away_ppg = sum({'W': 3, 'D': 1, 'L': 0}.get(m.get('result'), 0)
                               for m in away_matches) / len(away_matches) if away_matches else home_ppg
                home_advantage = home_ppg / max(away_ppg, 0.1)

        features = TeamFeatures(
            points_per_game=(wins * 3 + draws) / total,
            win_rate=wins / total,
            draw_rate=draws / total,
            loss_rate=losses / total,
            avg_goals_for=goals_for / total,
            avg_shots=avg_shots,
            avg_shots_on_target=avg_sot,
            finishing_efficiency=finishing_eff,
            xg_for_avg=xg_for_avg,
            avg_goals_against=goals_against / total,
            xg_against_avg=xg_against_avg,
            defensive_efficiency=defensive_eff,
            avg_possession=avg_possession,
            avg_corners=avg_corners,
            recent_points=recent_points,
            recent_goals_for=recent_gf,
            recent_goals_against=recent_ga,
            trend=trend,
            home_advantage=home_advantage if is_home else 1.0,
            away_disadvantage=away_disadvantage if not is_home else 1.0
        )
        features.is_home = is_home

        return features

    def _calculate_trend_value(self, recent_matches: List[Dict]) -> float:
        """Рассчитать тренд как число от -1 до 1"""
        if len(recent_matches) < 3:
            return 0.0

        # Разбиваем пополам
        mid = len(recent_matches) // 2
        first = recent_matches[:mid]
        second = recent_matches[mid:]

        def calc_ppg(matches):
            return sum({'W': 3, 'D': 1, 'L': 0}.get(m.get('result'), 0) for m in matches) / len(matches)

        ppg_first = calc_ppg(first)
        ppg_second = calc_ppg(second)

        # Нормализуем к -1..1
        diff = ppg_second - ppg_first
        return max(-1.0, min(1.0, diff / 1.5))

    def analyze_match(self, match_data: Dict, home_stats: Dict, away_stats: Dict,
                      h2h: Optional[Dict] = None) -> Tuple[Dict, MatchFeatures]:
        """Полный анализ с извлечением признаков для ML"""
        home_team = match_data.get('homeTeam', {}) or {}
        away_team = match_data.get('awayTeam', {}) or {}

        home_name = home_team.get('name', 'Home')
        away_name = away_team.get('name', 'Away')
        league_name = match_data.get('league', {}).get('name', '').lower()

        league_context = self._detect_league_context(league_name)

        # Извлекаем признаки команд
        home_features = self.extract_team_features(home_stats, True, league_context)
        away_features = self.extract_team_features(away_stats, False, league_context)

        # Создаем полный набор признаков матча
        match_features = MatchFeatures(
            home=home_features,
            away=away_features,
            h2h_home_wins=h2h.get('team1_wins', 0) if h2h else 0,
            h2h_draws=h2h.get('draws', 0) if h2h else 0,
            h2h_away_wins=h2h.get('team2_wins', 0) if h2h else 0,
            h2h_avg_goals=h2h.get('avg_goals', 2.5) if h2h else 2.5,
            league_avg_goals=league_context['avg_goals'],
            league_home_advantage=league_context['home_advantage']
        )

        # Базовый статистический прогноз (Пуассон)
        home_xg = self._calculate_xg(home_features, True, league_context)
        away_xg = self._calculate_xg(away_features, False, league_context)
        base_probs = self._calculate_probabilities(home_xg, away_xg)

        # ML прогноз
        ml_prediction = self.ml_model.predict(match_features)

        # Комбинированный прогноз
        final_probs = self._combine_predictions(base_probs, ml_prediction)

        # Генерация отчета
        analysis = {
            'home_name': home_name,
            'away_name': away_name,
            'home_xg': round(home_xg, 2),
            'away_xg': round(away_xg, 2),
            'probabilities': final_probs,
            'base_probabilities': base_probs,
            'ml_prediction': ml_prediction,
            'match_features': match_features,
            'narrative': self._generate_narrative(
                home_name, away_name, home_features, away_features,
                home_xg, away_xg, final_probs, ml_prediction, league_context, h2h
            ),
            'betting_tips': self._generate_betting_tips(final_probs, home_features, away_features),
            'key_factors': self._identify_key_factors(home_features, away_features, h2h),
            'confidence_score': self._calculate_confidence(home_features, away_features, ml_prediction),
            'league_context': league_context
        }

        return analysis, match_features

    def _detect_league_context(self, league_name: str) -> Dict:
        league_lower = league_name.lower()
        for league_key, context in self.league_context.items():
            if league_key.replace('_', ' ') in league_lower or league_key in league_lower:
                return context
        return {'avg_goals': 2.5, 'home_advantage': 1.4, 'style': 'balanced'}

    def _calculate_xg(self, features: TeamFeatures, is_home: bool,
                      league_context: Dict) -> float:
        """Рассчитать ожидаемые голы"""
        base = features.xg_for_avg if features.xg_for_avg > 0 else features.avg_goals_for

        if is_home:
            base *= league_context['home_advantage'] * features.home_advantage
        else:
            base *= 0.85 * features.away_disadvantage

        # Корректировки
        form_mult = 1 + (features.trend * 0.15)
        eff_mult = 0.5 + (features.finishing_efficiency * 0.5)
        poss_mult = 0.8 + (features.avg_possession / 100) * 0.4

        final = base * form_mult * eff_mult * poss_mult
        return max(0.3, min(final, 3.5))

    def _calculate_probabilities(self, home_xg: float, away_xg: float) -> Dict:
        from scipy.stats import poisson

        max_goals = 6
        score_probs = {}

        for h in range(max_goals + 1):
            for a in range(max_goals + 1):
                prob = poisson.pmf(h, home_xg) * poisson.pmf(a, away_xg)
                score_probs[f"{h}-{a}"] = prob

        home_win = sum(p for s, p in score_probs.items() if int(s.split('-')[0]) > int(s.split('-')[1]))
        draw = sum(p for s, p in score_probs.items() if int(s.split('-')[0]) == int(s.split('-')[1]))
        away_win = sum(p for s, p in score_probs.items() if int(s.split('-')[0]) < int(s.split('-')[1]))

        total = home_win + draw + away_win

        most_likely = max(score_probs, key=score_probs.get)

        return {
            'home_win': round(home_win / total * 100, 1),
            'draw': round(draw / total * 100, 1),
            'away_win': round(away_win / total * 100, 1),
            'most_likely_score': most_likely,
            'most_likely_prob': round(score_probs[most_likely] * 100, 1),
            'over_2_5': round(sum(p for s, p in score_probs.items()
                                  if int(s.split('-')[0]) + int(s.split('-')[1]) > 2.5) * 100, 1),
            'btts': round(sum(p for s, p in score_probs.items()
                              if int(s.split('-')[0]) > 0 and int(s.split('-')[1]) > 0) * 100, 1),
            'expected_total': round(home_xg + away_xg, 2)
        }

    def _combine_predictions(self, base_probs: Dict,
                             ml_probs: Optional[Dict]) -> Dict:
        """Умное комбинирование статистики и ML"""
        if not ml_probs or ml_probs['reliability'] == 'low':
            # ML не уверен - доверяем статистике
            weight_ml = 0.1
        elif ml_probs['reliability'] == 'medium':
            weight_ml = 0.25
        else:
            weight_ml = 0.4

        weight_base = 1 - weight_ml

        combined = {}
        for key in ['home_win', 'draw', 'away_win']:
            base = base_probs.get(key, 33)
            ml = ml_probs.get(key, 33) if ml_probs else 33
            combined[key] = base * weight_base + ml * weight_ml

        # Нормализация
        total = sum(combined.values())
        for key in combined:
            combined[key] = round(combined[key] / total * 100, 1)

        # Копируем остальные поля из базового прогноза
        for key in ['most_likely_score', 'most_likely_prob', 'over_2_5', 'btts', 'expected_total']:
            combined[key] = base_probs[key]

        combined['ml_weight_used'] = weight_ml
        return combined

    def _generate_narrative(self, home_name: str, away_name: str,
                            home_f: TeamFeatures, away_f: TeamFeatures,
                            home_xg: float, away_xg: float,

                         f"надежность {ml_pred['reliability']}\n")

        # Хозяева
        lines.append(f"\n🏠 {home_name} (дома)")

        lines.append(f"  Оборона: {home_f.avg_goals_against:.2f} гола пропущено")
        lines.append(f"  Владение: {home_f.avg_possession:.1f}%, удары: {home_f.avg_shots:.1f}")

        # Гости
        lines.append(f"\n✈️ {away_name} (в гостях)")
        lines.append(f"  Форма: {away_f.points_per_game:.2f} очка/матч, тренд {away_f.trend:+.2f}")
        lines.append(f"  Атака: {away_f.avg_goals_for:.2f} гола, xG {away_f.xg_for_avg:.2f}")
        lines.append(f"  Оборона: {away_f.avg_goals_against:.2f} гола пропущено")

        # H2H
        if h2h and h2h.get('total_matches', 0) > 0:
            lines.append(f"\n📊 Личные встречи: {h2h['team1_wins']}-{h2h['draws']}-{h2h['team2_wins']}, "
                         f"средний голы {h2h['avg_goals']:.1f}")

        # Прогноз
        lines.append(f"\n🎯 Прогноз:")
        lines.append(f"  П1: {probs['home_win']}% | X: {probs['draw']}% | П2: {probs['away_win']}%")
        lines.append(f"  Счет: {probs['most_likely_score']} (вероятность {probs['most_likely_prob']}%)")
        lines.append(f"  ТБ 2.5: {probs['over_2_5']}%, Обе забьют: {probs['btts']}%")

        # Вывод
        lines.append(f"\n💡 Вывод:")
        if probs['home_win'] > 55:
            lines.append(f"  {home_name} фаворит при домашнем преимуществе")
        elif probs['away_win'] > 55:
            lines.append(f"  {away_name} фаворит несмотря на выезд")
        elif abs(probs['home_win'] - probs['away_win']) < 10:
            lines.append(f"  Равный матч, вероятна ничья или минимальная победа")
        else:
            fav = home_name if probs['home_win'] > probs['away_win'] else away_name
            lines.append(f"  Небольшое преимущество у {fav}")

        return "\n".join(lines)

    def _generate_betting_tips(self, probs: Dict, home_f: TeamFeatures,
                               away_f: TeamFeatures) -> List[Dict]:
        tips = []

        # Основной исход
        if probs['home_win'] > 60:
            tips.append({
                'market': f"П1 ({probs['home_win']}%)",
                'confidence': 'high',
                'reason': 'Сильные хозяева' if home_f.win_rate > 0.5 else 'Домашнее преимущество'
            })
        elif probs['away_win'] > 55:
            tips.append({
                'market': f"П2 ({probs['away_win']}%)",
                'confidence': 'medium',
                'reason': 'Гости в форме' if away_f.trend > 0 else 'Слабые хозяева'
            })

        # Тоталы
        if probs['expected_total'] > 2.8:
            tips.append({
                'market': f"ТБ 2.5 ({probs['over_2_5']}%)",
                'confidence': 'high' if probs['over_2_5'] > 65 else 'medium',
                'reason': f"Высокая результативность ожидается"
            })
        elif probs['expected_total'] < 2.2:
            tips.append({
                'market': f"ТМ 2.5 ({100 - probs['over_2_5']:.1f}%)",
                'confidence': 'medium',
                'reason': "Закрытый матч ожидается"
            })

        # Обе забьют
        if probs['btts'] > 60:
            tips.append({
                'market': f"Обе забьют ДА ({probs['btts']}%)",
                'confidence': 'medium',
                'reason': "Обе команды атакующие"
            })

        return tips

    def _identify_key_factors(self, home_f: TeamFeatures, away_f: TeamFeatures,
                              h2h: Optional[Dict]) -> List[str]:
        factors = []

        if home_f.trend > 0.3:
            factors.append(f"📈 {home_f.home_advantage:.2f}x домашнее преимущество")
        if away_f.trend > 0.3:
            factors.append(f"📈 Гости на подъеме")
        if home_f.finishing_efficiency > 1.2:
            factors.append("🎯 Хозяева клинически реализуют моменты")
        if away_f.defensive_efficiency < 0.9:
            factors.append("🛡️ Прочная оборона гостей")
        if h2h and h2h.get('team1_wins', 0) > h2h.get('team2_wins', 0) * 2:
            factors.append("📊 Историческое преимущество хозяев")

        return factors if factors else ["Стандартный матч без явных факторов"]

    def _calculate_confidence(self, home_f: TeamFeatures, away_f: TeamFeatures,
                              ml_pred: Optional[Dict]) -> int:
        conf = 50

        # Достоверность данных
        if home_f.win_rate != 0.33:  # не дефолт
            conf += 15
        if away_f.win_rate != 0.33:
            conf += 15

        # ML уверенность
        if ml_pred:
            conf += min(ml_pred['confidence'] / 5, 15)

        # Стабильность
        if abs(home_f.trend) < 0.3:
            conf += 5
        if abs(away_f.trend) < 0.3:
            conf += 5

        return min(conf, 95)

        self.api_key = api_key
        self.base_url = SSTATS_BASE_URL
        self.session = requests.Session()
        self.session.headers.update(HEADERS_SSTATS)
        self.last_request_time = 0
        self.min_delay = 0.15
        self.ai_analyzer = AIAnalyzer()

    def _rate_limit(self):
        current_time = time.time()
        elapsed = current_time - self.last_request_time
        if elapsed < self.min_delay:
            time.sleep(self.min_delay - elapsed)
        self.last_request_time = time.time()

    def _make_request(self, endpoint: str, params: Dict = None) -> Optional[Dict]:
        self._rate_limit()
        url = f"{self.base_url}{endpoint}"
        params = params or {}
        params['apikey'] = self.api_key

        try:
            response = self.session.get(url, params=params, timeout=30)
            response.raise_for_status()
            return response.json()
        except Exception as e:
            print(f"API Error {endpoint}: {e}")
            return None

    @cache_api_request(ttl_seconds=3600)
    def get_leagues(self) -> List[Dict]:
        data = self._make_request("/leagues")
        return data.get('data', []) if data else []

    def get_all_matches_for_date(self, date: str, league_id: int = None) -> List[Dict]:
        all_matches = []
        offset = 0
        limit = 1000

        while True:
            params = {'limit': limit, 'offset': offset}
            if date:
                params['date'] = date
            if league_id:
                params['leagueid'] = league_id

            data = self._make_request("/games/list", params)
            matches = data.get('data', []) if data else []

            if not matches:
                break

            all_matches.extend(matches)

            if len(matches) < limit:
                break

            offset += limit
            if offset > 15000:
                break

        return all_matches

    @cache_api_request(ttl_seconds=1800)
    def get_match_details(self, match_id) -> Optional[Dict]:
        data = self._make_request(f"/games/{match_id}")
        return data.get('data') if data else None

    @cache_api_request(ttl_seconds=900)
    def get_team_last_matches(self, team_id: int, limit: int = 10) -> List[Dict]:
        params = {
            'team': team_id,
            'ended': 'true',
            'order': -1,
            'limit': limit
        }

        data = self._make_request('/games/list', params)

        if not data:
            return []

        matches = data.get('data', []) if isinstance(data, dict) else data

        if not matches:
            return []

        detailed_matches = []
        with ThreadPoolExecutor(max_workers=3) as executor:
            future_to_match = {
                executor.submit(self.get_match_details, match.get('id')): match
                for match in matches if match.get('id')
            }

            for future in as_completed(future_to_match):
                match = future_to_match[future]
                try:
                    details = future.result()
                    if details:
                        full_match = {**match, **details}
                        detailed_matches.append(full_match)
                    else:
                        detailed_matches.append(match)
                except:
                    detailed_matches.append(match)

        return sorted(detailed_matches, key=lambda x: x.get('date', ''), reverse=True)

    def extract_team_stats(self, match_data: Dict, team_id: int) -> Dict:
        """Извлечь статистику с защитой от пустых данных"""
        stats = {
            'goals_for': 0, 'goals_against': 0,
            'xg_for': None, 'xg_against': None,
            'possession': 50, 'shots': 0, 'shots_on_target': 0,
            'corners': 0, 'date': match_data.get('date', ''),
            'opponent': '', 'is_home': False, 'result': ''
        }

        home_team = match_data.get('homeTeam', {}) or {}
        away_team = match_data.get('awayTeam', {}) or {}

        # Проверяем, что данные команд существуют
        if not isinstance(home_team, dict) or not isinstance(away_team, dict):
            return stats

        home_id = home_team.get('id')
        away_id = away_team.get('id')

        if home_id == team_id:
            stats['is_home'] = True
            stats['opponent'] = away_team.get('name', 'Unknown')
            stats['goals_for'] = match_data.get('homeResult', 0) or 0
            stats['goals_against'] = match_data.get('awayResult', 0) or 0
        elif away_id == team_id:
            stats['is_home'] = False
            stats['opponent'] = home_team.get('name', 'Unknown')
            stats['goals_for'] = match_data.get('awayResult', 0) or 0
            stats['goals_against'] = match_data.get('homeResult', 0) or 0
        else:
            return stats

        # Определяем результат
        if stats['goals_for'] > stats['goals_against']:
            stats['result'] = 'W'
        elif stats['goals_for'] < stats['goals_against']:
            stats['result'] = 'L'
        else:
            stats['result'] = 'D'

        # xG данные (могут отсутствовать)
        xg_data = match_data.get('xg', {})
        if isinstance(xg_data, dict):
            if stats['is_home']:
                stats['xg_for'] = xg_data.get('home')
                stats['xg_against'] = xg_data.get('away')
            else:
                stats['xg_for'] = xg_data.get('away')
                stats['xg_against'] = xg_data.get('home')

        # Статистика матча (может отсутствовать для малых лиг)
        stats_data = match_data.get('statistics', {})
        if isinstance(stats_data, dict):
            prefix = 'Home' if stats['is_home'] else 'Away'

            # Пытаемся получить статистику, но не критично если её нет
            poss_key = f'ballPossession{prefix}'
            if poss_key in stats_data:
                try:
                    stats['possession'] = float(stats_data[poss_key])
                except:
                    pass

            shots_key = f'totalShots{prefix}'
            if shots_key in stats_data:
                try:
                    stats['shots'] = int(stats_data[shots_key])
                except:
                    pass

            sot_key = f'shotsOnGoal{prefix}'
            if sot_key in stats_data:
                try:
                    stats['shots_on_target'] = int(stats_data[sot_key])
                except:
                    pass

            corners_key = f'cornerKicks{prefix}'
            if corners_key in stats_data:
                try:
                    stats['corners'] = int(stats_data[corners_key])
                except:
                    pass

        return stats

    @cache_api_request(ttl_seconds=1800)
    def get_team_full_stats(self, team_id: int, team_name: str = "") -> Dict:
        """Получить полную статистику команды с защитой от ошибок"""
        try:
            matches = self.get_team_last_matches(team_id, limit=10)
        except Exception as e:
            print(f"[API] Error getting matches for team {team_id}: {e}")
            matches = []

        if not matches:
            return {
                'matches': [],
                'team_id': team_id,
                'team_name': team_name,
                'total_matches': 0,
                'has_data': False
            }

        stats_list = []
        for match in matches:
            try:
                match_stats = self.extract_team_stats(match, team_id)
                if match_stats and match_stats.get('result'):
                    stats_list.append(match_stats)
            except Exception as e:
                print(f"[API] Error extracting stats from match: {e}")
                continue

        return {
            'matches': stats_list,
            'team_id': team_id,
            'team_name': team_name,
            'total_matches': len(stats_list),
            'has_data': len(stats_list) > 0
        }

    @cache_api_request(ttl_seconds=3600)
    def get_head_to_head(self, team1_id: int, team2_id: int, limit: int = 5) -> Optional[Dict]:
        params = {
            'team1': team1_id,
            'team2': team2_id,
            'ended': 'true',
            'limit': limit
        }

        data = self._make_request('/games/headtohead', params)
        matches = data.get('data', []) if data else []

        if not matches:
            return None

        team1_wins = team2_wins = draws = 0
        total_goals = 0

        for match in matches:
            home_res = match.get('homeResult', 0) or 0
            away_res = match.get('awayResult', 0) or 0

            home_id = (match.get('homeTeam') or {}).get('id')

            if home_res == away_res:
                draws += 1
            elif (home_res > away_res and home_id == team1_id) or \
                    (away_res > home_res and home_id != team1_id):
                team1_wins += 1
            else:
                team2_wins += 1

            total_goals += home_res + away_res

        return {
            'matches': matches,
            'team1_wins': team1_wins,
            'team2_wins': team2_wins,
            'draws': draws,
            'total_matches': len(matches),
            'avg_goals': total_goals / len(matches) if matches else 0
        }

    def analyze_match_with_ai(self, match_data: Dict) -> Dict:
        """Полный анализ с защитой от отсутствия данных"""
        home_team = match_data.get('homeTeam', {}) or {}
        away_team = match_data.get('awayTeam', {}) or {}

        home_id = home_team.get('id')
        away_id = away_team.get('id')
        home_name = home_team.get('name', 'Home')
        away_name = away_team.get('name', 'Away')

        if not home_id or not away_id:
            return {'error': 'Missing team IDs'}

        # Параллельная загрузка с таймаутом
        with ThreadPoolExecutor(max_workers=3) as executor:
            future_home = executor.submit(self.get_team_full_stats, home_id, home_name)
            future_away = executor.submit(self.get_team_full_stats, away_id, away_name)
            future_h2h = executor.submit(self.get_head_to_head, home_id, away_id)

            try:
                home_stats = future_home.result(timeout=10)
                away_stats = future_away.result(timeout=10)
                h2h = future_h2h.result(timeout=10)
            except Exception as e:
                print(f"[API] Timeout or error: {e}")
                home_stats = {'matches': [], 'total_matches': 0}
                away_stats = {'matches': [], 'total_matches': 0}
                h2h = None

        # Проверяем, есть ли данные
        has_data = (home_stats.get('total_matches', 0) > 0 or
                    away_stats.get('total_matches', 0) > 0)

        if not has_data:
            print(f"[ANALYSIS] No data for {home_name} vs {away_name}, using basic analysis")
            return self._basic_analysis(match_data, home_name, away_name)

        # AI анализ + извлечение признаков
        try:
            analysis, match_features = self.ai_analyzer.analyze_match(
                match_data, home_stats, away_stats, h2h
            )

            # ML предсказание отдельно
            ml_prediction = self.ai_analyzer.ml_model.predict(match_features)

            # Сохраняем с полными признаками
            db.save_prediction_with_features(match_data, analysis, match_features, ml_prediction)

            return analysis
        except Exception as e:
            print(f"[ANALYSIS] AI analysis error: {e}")
            return self._basic_analysis(match_data, home_name, away_name)

    def _basic_analysis(self, match_data: Dict, home_name: str, away_name: str) -> Dict:
        """Базовый анализ для матчей без подробной статистики"""

        # Пытаемся получить базовую информацию из самого match_data
        league = match_data.get('league', {})
        league_name = league.get('name', 'Unknown')

        # Дефолтные вероятности
        probs = {
            'home_win': 40.0,
            'draw': 25.0,
            'away_win': 35.0,
            'most_likely_score': '1-1',
            'most_likely_prob': 12.0,
            'over_2_5': 45.0,
            'btts': 50.0,
            'expected_total': 2.3
        }

        # Если есть коэффициенты в данных - используем их
        odds = match_data.get('odds', {})
        if odds:
            try:
                home_odd = float(odds.get('home', 2.0))
                draw_odd = float(odds.get('draw', 3.2))
                away_odd = float(odds.get('away', 3.5))

                # Конвертируем коэффициенты в вероятности
                total_prob = (1/home_odd + 1/draw_odd + 1/away_odd)
                probs['home_win'] = round((1/home_odd) / total_prob * 100, 1)
                probs['draw'] = round((1/draw_odd) / total_prob * 100, 1)
                probs['away_win'] = round((1/away_odd) / total_prob * 100, 1)
            except Exception as e:
                print(f"[ANALYSIS] Error parsing odds: {e}")

        narrative = f"""📊 Базовый анализ: {home_name} vs {away_name}

⚠️ Ограниченные данные
Для этого матча недостаточно статистики в базе sstats.net.
Возможные причины:
• Матч из низшей лиги или регионального турнира
• Команды мало играли в последнее время
• Данные еще не обновлены

🎯 Прогноз на основе доступных данных:
П1: {probs['home_win']}% | X: {probs['draw']}% | П2: {probs['away_win']}%

💡 Рекомендация:
Без детальной статистики точность прогноза снижена.
Рекомендуется изучить текущую форму команд самостоятельно
или выбрать матч из топ-5 лиг для полного анализа.
"""

        return {
            'home_name': home_name,
            'away_name': away_name,
            'home_xg': 1.2,
            'away_xg': 1.1,
            'probabilities': probs,
            'base_probabilities': probs,
            'ml_prediction': None,
            'match_features': None,
            'narrative': narrative,
            'betting_tips': [{
                'market': 'Недостаточно данных',
                'confidence': 'low',
                'reason': 'Нет статистики команд'
            }],
            'key_factors': ['⚠️ Нет данных о последних матчах'],
            'confidence_score': 20,
            'league_context': {'avg_goals': 2.5, 'home_advantage': 1.4, 'style': 'unknown'}
        }

    def save_prediction_json(self, analysis: Dict):
        desktop = self.get_desktop_path()

        home = analysis['home_name'].replace(' ', '_')
        away = analysis['away_name'].replace(' ', '_')
        date = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"Прогноз_{home}_vs_{away}_{date}.json"

        filepath = desktop / filename

        data = {
            'match': f"{analysis['home_name']} vs {analysis['away_name']}",
            'date_created': datetime.now().isoformat(),
            'probabilities': analysis['probabilities'],
            'ml_prediction': analysis.get('ml_prediction'),
            'expected_goals': {
                'home': analysis['home_xg'],
                'away': analysis['away_xg']
            },
            'narrative': analysis['narrative'],
            'betting_tips': analysis.get('betting_tips', []),
            'key_factors': analysis.get('key_factors', []),
            'confidence_score': analysis.get('confidence_score', 0)

            ["", ""],
            ["Счет", probs['most_likely_score']],
            ["ТБ 2.5", f"{probs['over_2_5']}%"],
        ]

        for i, row in enumerate(data, start=4):
            for j, val in enumerate(row, start=1):
                ws.cell(row=i, column=j, value=val)

        # ML
        if analysis.get('ml_prediction'):
            ws['A12'] = "ML МОДЕЛЬ"
            ws['A12'].font = header_font
            ws['A12'].fill = header_fill

            ml = analysis['ml_prediction']
            ml_data = [

    def handle_save(self, format_type: str, analysis: Dict):
        try:
            if format_type == 'json':
                path = self.save_prediction_json(analysis)
            elif format_type == 'txt':
                path = self.save_prediction_txt(analysis)
            elif format_type == 'excel':
                path = self.save_prediction_excel(analysis)
            else:
                return

            if path:
                messagebox.showinfo("Успех", f"Прогноз сохранен:\n{path.name}")
        except Exception as e:
            messagebox.showerror("Ошибка сохранения", str(e))

    def set_quick_date(self, days: int):
        date = datetime.now() + timedelta(days=days)
        self.selected_date = date
        if CALENDAR_AVAILABLE:
            self.calendar.selection_set(date.strftime('%Y-%m-%d'))
        self.load_matches()

    def on_date_changed(self, event=None):
        if CALENDAR_AVAILABLE:
            date_str = self.calendar.get_date()
            self.selected_date = datetime.strptime(date_str, '%Y-%m-%d')
        self.load_matches()

    def load_matches(self):
        self.status_bar.config(text="Загрузка матчей...")

        def fetch():
            try:
                date_str = self.selected_date.strftime("%Y-%m-%d")
                league_name = self.league_var.get()

                league_id = None
                for league in self.leagues:
                    if league.get('name') == league_name:
                        league_id = league.get('id')
                        break

                matches = self.api.get_all_matches_for_date(date_str, league_id)
                self.root.after(0, lambda: self.on_matches_loaded(matches))
            except Exception as e:
                self.root.after(0, lambda: self.on_error(str(e)))

        threading.Thread(target=fetch, daemon=True).start()

    def on_matches_loaded(self, matches: List[Dict]):
        self.current_matches = matches

        for item in self.matches_tree.get_children():
            self.matches_tree.delete(item)

        for match in matches:
            try:
                date_str = match.get('date', '')
                time_str = "--:--"
                if date_str:
                    try:
                        dt = datetime.fromisoformat(date_str.replace('Z', '+00:00'))
                        time_str = dt.strftime("%H:%M")
                    except:
                        pass

                league = match.get('league', {}).get('name', 'Unknown') if isinstance(match.get('league'),
                                                                                      dict) else 'Unknown'
                home = match.get('homeTeam', {}).get('name', 'Home') if isinstance(match.get('homeTeam'),
                                                                                   dict) else 'Home'
                away = match.get('awayTeam', {}).get('name', 'Away') if isinstance(match.get('awayTeam'),
                                                                                   dict) else 'Away'

                self.matches_tree.insert('', tk.END, values=(
                    time_str, league, home, away, "Анализ"
                ), tags=(str(match.get('id')),))

            except Exception as e:
                print(f"Error inserting match: {e}")

        self.stats_label.config(
            text=f"Загружено: {len(matches)} матчей\nДата: {self.selected_date.strftime('%d.%m.%Y')}")
        self.status_bar.config(text=f"Готово | {len(matches)} матчей")

    def filter_matches(self, *args):
        search_text = self.search_var.get().lower()

        for item in self.matches_tree.get_children():
            self.matches_tree.delete(item)

        for match in self.current_matches:
            home = match.get('homeTeam', {}).get('name', '').lower() if isinstance(match.get('homeTeam'), dict) else ''
            away = match.get('awayTeam', {}).get('name', '').lower() if isinstance(match.get('awayTeam'), dict) else ''
            league = match.get('league', {}).get('name', '').lower() if isinstance(match.get('league'), dict) else ''

            if search_text and search_text not in home and search_text not in away and search_text not in league:
                continue

            try:
                date_str = match.get('date', '')
                time_str = "--:--"
                if date_str:
                    try:
                        dt = datetime.fromisoformat(date_str.replace('Z', '+00:00'))
                        time_str = dt.strftime("%H:%M")
                    except:
                        pass

                league_name = match.get('league', {}).get('name', 'Unknown') if isinstance(match.get('league'),
                                                                                           dict) else 'Unknown'
                home_name = match.get('homeTeam', {}).get('name', 'Home') if isinstance(match.get('homeTeam'),
                                                                                        dict) else 'Home'
                away_name = match.get('awayTeam', {}).get('name', 'Away') if isinstance(match.get('awayTeam'),
                                                                                        dict) else 'Away'

                self.matches_tree.insert('', tk.END, values=(
                    time_str, league_name, home_name, away_name, "Анализ"
                ), tags=(str(match.get('id')),))
            except:
                pass

    def on_match_select(self, event):
        item = self.matches_tree.selection()
        if not item:
            return

        tags = self.matches_tree.item(item[0], 'tags')
        if not tags:
            return

        match_id = tags[0]
        match = next((m for m in self.current_matches if str(m.get('id')) == match_id), None)

        if not match:
            return

        self.show_ai_analysis(match)

    def train_ml_model(self):
        """Обучить ML модель на исторических данных"""
        if not SKLEARN_AVAILABLE:
            messagebox.showerror("Ошибка", "scikit-learn не установлен")
            return

        # Получаем данные для обучения
        training_data = db.get_training_data(min_confidence=0)

        if len(training_data) < 30:
            messagebox.showwarning("Внимание",
                                   f"Недостаточно данных: {len(training_data)} матчей\n"
                                   f"Нужно минимум 30 завершенных матчей с сохраненными признаками.\n"
                                   f"Продолжайте делать прогнозы и обновляйте результаты.")
            return

        result = self.api.ai_analyzer.ml_model.train(training_data)

        if not result['success']:
            messagebox.showerror("Ошибка обучения", result.get('error', 'Неизвестная ошибка'))
            return

        msg = (f"Обучение завершено успешно!\n\n"
               f"Точность: {result['accuracy']:.1f}%\n"
               f"CV mean: {result['cv_mean']:.1f}% (+/- {result['cv_std']:.1f}%)\n"

    def check_results(self):
        """
        Проверить и обновить результаты матчей (автоматический режим):
        - Берём дату, выбранную в календаре
        - Запрашиваем матчи дня у API
        - Обновляем фактические счета и prediction_history
        """
        if not self.current_matches:
            messagebox.showwarning("Внимание", "Нет загруженных матчей для выбранной даты")
            return

        date_str = self.selected_date.strftime("%Y-%m-%d")
        self.status_bar.config(text=f"Обновление результатов за {date_str}...")

        def worker():
            try:
                # можно привязать к конкретной лиге, если нужно
                league_id = None
                league_name = self.league_var.get()
                for league in self.leagues:
                    if league.get('name') == league_name:
                        league_id = league.get('id')
                        break

                self.api.update_matches_results_for_date(date_str, league_id)
                self.root.after(0, lambda: (
                    self.status_bar.config(text="Результаты обновлены"),
                    messagebox.showinfo("Готово",
                                        "Результаты завершенных матчей обновлены.\n"
                                        "Теперь можно обучать ML модель.")
                ))
            except Exception as e:
                self.root.after(0, lambda: (
                    self.status_bar.config(text="Ошибка обновления результатов"),
                    messagebox.showerror("Ошибка", str(e))
                ))

        threading.Thread(target=worker, daemon=True).start()

    def export_daily_predictions(self):
        """Экспорт всех прогнозов дня"""
        if not self.current_matches:
            messagebox.showwarning("Внимание", "Нет загруженных матчей")
            return

        if not EXCEL_AVAILABLE:
            messagebox.showerror("Ошибка", "Excel экспорт недоступен")
            return

        self.status_bar.config(text="Создание отчета...")

        def export():
            try:
                desktop = self.get_desktop_path()
                date_str = self.selected_date.strftime("%Y%m%d")
                filename = f"Все_прогнозы_{date_str}.xlsx"
                filepath = desktop / filename

                wb = Workbook()
                ws = wb.active
                ws.title = "Прогнозы"

                headers = ['Время', 'Лига', 'Хозяева', 'Гости', 'П1%', 'X%', 'П2%',
                           'ML П1', 'ML X', 'ML П2', 'Счет', 'xG Хоз', 'xG Гост', 'Уверенность']
                for col, header in enumerate(headers, 1):
                    cell = ws.cell(row=1, column=col, value=header)
                    cell.font = Font(bold=True)
                    cell.fill = PatternFill(start_color="3b82f6", fill_type="solid")
                    cell.font = Font(bold=True, color="FFFFFF")

                row = 2
                for match in self.current_matches:
                    try:
                        analysis = self.api.analyze_match_with_ai(match)
                        if 'error' in analysis:
                            continue

                        probs = analysis['probabilities']
                        ml = analysis.get('ml_prediction', {})

                        ws.cell(row=row, column=1, value=match.get('date', '')[11:16] if match.get('date') else '--:--')
                        ws.cell(row=row, column=2, value=(match.get('league') or {}).get('name', ''))
                        ws.cell(row=row, column=3, value=analysis['home_name'])
                        ws.cell(row=row, column=4, value=analysis['away_name'])
                        ws.cell(row=row, column=5, value=probs['home_win'])
                        ws.cell(row=row, column=6, value=probs['draw'])
                        ws.cell(row=row, column=7, value=probs['away_win'])
                        ws.cell(row=row, column=8, value=ml.get('home_win', '-'))
                        ws.cell(row=row, column=9, value=ml.get('draw', '-'))
                        ws.cell(row=row, column=10, value=ml.get('away_win', '-'))
                        ws.cell(row=row, column=11, value=probs['most_likely_score'])
                        ws.cell(row=row, column=12, value=analysis['home_xg'])
                        ws.cell(row=row, column=13, value=analysis['away_xg'])
                        ws.cell(row=row, column=14, value=analysis.get('confidence_score', 0))

                        row += 1
                    except:
                        continue

                for col in ws.columns:
                    max_length = 0
                    column = col[0].column_letter
                    for cell in col:
                        try:
                            if len(str(cell.value)) > max_length:
                                max_length = len(str(cell.value))
                        except:
                            pass
                    adjusted_width = min(max_length + 2, 50)
                    ws.column_dimensions[column].width = adjusted_width

                wb.save(filepath)

                self.root.after(0, lambda: messagebox.showinfo(
                    "Успех", f"Отчет сохранен:\n{filepath.name}\n\nМатчей: {row - 2}"
                ))
                self.root.after(0, lambda: self.status_bar.config(text="Готово"))

            except Exception as e:
                self.root.after(0, lambda: messagebox.showerror("Ошибка", str(e)))
                self.root.after(0, lambda: self.status_bar.config(text="Ошибка экспорта"))

        threading.Thread(target=export, daemon=True).start()

    def setup_notifications(self):
        """Настроить уведомления"""
        if not self.current_matches:
            messagebox.showwarning("Внимание", "Нет загруженных матчей")
            return

        window = tk.Toplevel(self.root)
        window.title("Уведомления")
        window.geometry("400x300")

        tk.Label(window, text="Выберите матчи для уведомлений:",
                 font=('Segoe UI', 12, 'bold')).pack(pady=10)

        frame = ttk.Frame(window)
        frame.pack(fill=tk.BOTH, expand=True, padx=10, pady=5)

        self.notification_vars = {}

        for match in self.current_matches[:10]:
            home = (match.get('homeTeam') or {}).get('name', 'Home')
            away = (match.get('awayTeam') or {}).get('name', 'Away')
            match_time = match.get('date', '')[11:16] if match.get('date') else '--:--'

            var = tk.BooleanVar(value=False)
            self.notification_vars[match.get('id')] = var

            text = f"{match_time} - {home} vs {away}"
            ttk.Checkbutton(frame, text=text, variable=var).pack(anchor='w', pady=2)

        ttk.Button(window, text="Запланировать",
                   command=self.schedule_selected_notifications).pack(pady=10)

    def schedule_selected_notifications(self):
        """Запланировать выбранные уведомления"""
        scheduled = 0

        for match_id, var in self.notification_vars.items():
            if var.get():
                match = next((m for m in self.current_matches if m.get('id') == match_id), None)
                if match:
                    self.schedule_notification(match)
                    scheduled += 1

        messagebox.showinfo("Успех", f"Запланировано уведомлений: {scheduled}")

    def schedule_notification(self, match: Dict):
        """Запланировать уведомление"""
        try:
            match_time = datetime.fromisoformat(match['date'].replace('Z', '+00:00'))
            notify_time = match_time - timedelta(minutes=30)

            delay = (notify_time - datetime.now()).total_seconds()

            if delay > 0:
                def notify():
                    home = (match.get('homeTeam') or {}).get('name', 'Home')
                    away = (match.get('awayTeam') or {}).get('name', 'Away')

                    messagebox.showinfo(
                        "⚽ Матч начинается!",
                        f"{home} vs {away}\nНачало через 30 минут!"
                    )

                timer = threading.Timer(delay, notify)
                timer.daemon = True
                timer.start()
                self.notification_threads.append(timer)

        except Exception as e:
            print(f"Error scheduling notification: {e}")

        try:
            settings = db.execute("SELECT key, value FROM settings")
            for key, value in settings:
                if key == 'theme':
                    self.theme = value

    def load_initial_data(self):
        def fetch():
            try:
                leagues = self.api.get_leagues()
                self.root.after(0, lambda: self.on_leagues_loaded(leagues))
            except Exception as e:
                self.root.after(0, lambda: self.on_error(str(e)))

        threading.Thread(target=fetch, daemon=True).start()

    def on_leagues_loaded(self, leagues):
        self.leagues = leagues
        league_names = ["Все лиги"] + [l.get('name', 'Unknown') for l in leagues]
        self.league_combo['values'] = league_names
        self.load_matches()
        self.update_ml_status()

        messagebox.showerror("Ошибка", msg)

        if loading_window:
            loading_window.destroy()
        self.on_error(msg)
